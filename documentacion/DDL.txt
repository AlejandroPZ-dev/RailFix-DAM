-- Drop table

-- DROP TABLE public.lineas;

CREATE TABLE public.lineas (
	id_linea bigserial NOT NULL,
	codigo_linea varchar(20) NOT NULL,
	nombre varchar(150) NOT NULL,
	origen varchar(100) NOT NULL,
	destino varchar(100) NOT NULL,
	pk_inicial numeric(8, 3) NOT NULL,
	pk_final numeric(8, 3) NOT NULL,
	CONSTRAINT chk_lineas_pk CHECK ((pk_final > pk_inicial)),
	CONSTRAINT lineas_codigo_linea_key UNIQUE (codigo_linea),
	CONSTRAINT lineas_nombre_key UNIQUE (nombre),
	CONSTRAINT lineas_pkey PRIMARY KEY (id_linea)
);

-- Drop table

-- DROP TABLE public.usuarios;

CREATE TABLE public.usuarios (
	id_usuario bigserial NOT NULL,
	nombre varchar(100) NOT NULL,
	apellidos varchar(150) NOT NULL,
	username varchar(50) NOT NULL,
	password_hash varchar(255) NOT NULL,
	rol varchar(20) NOT NULL,
	activo bool DEFAULT true NOT NULL,
	fecha_creacion timestamp DEFAULT CURRENT_TIMESTAMP NOT NULL,
	CONSTRAINT usuarios_pkey PRIMARY KEY (id_usuario),
	CONSTRAINT usuarios_rol_check CHECK (((rol)::text = ANY ((ARRAY['OPERARIO'::character varying, 'ADMINISTRADOR'::character varying, 'TECNICO'::character varying])::text[]))),
	CONSTRAINT usuarios_username_key UNIQUE (username)
);


-- public.vias definition

-- Drop table

-- DROP TABLE public.vias;

CREATE TABLE public.vias (
	id_via bigserial NOT NULL,
	id_linea int8 NOT NULL,
	codigo_via varchar(20) NOT NULL,
	nombre varchar(150) NOT NULL,
	sentido varchar(20) NOT NULL,
	activa bool DEFAULT true NOT NULL,
	CONSTRAINT uk_vias_linea_codigo UNIQUE (id_linea, codigo_via),
	CONSTRAINT vias_pkey PRIMARY KEY (id_via),
	CONSTRAINT vias_sentido_check CHECK (((sentido)::text = ANY ((ARRAY['CRECIENTE'::character varying, 'DECRECIENTE'::character varying, 'AMBOS'::character varying, 'NO_APLICA'::character varying])::text[]))),
	CONSTRAINT fk_vias_lineas FOREIGN KEY (id_linea) REFERENCES public.lineas(id_linea)
);


-- public.incidencias definition

-- Drop table

-- DROP TABLE public.incidencias;

CREATE TABLE public.incidencias (
	id_incidencia bigserial NOT NULL,
	id_linea int8 NOT NULL,
	id_via int8 NOT NULL,
	id_operario_creador int8 NOT NULL,
	titulo varchar(150) NOT NULL,
	descripcion text NOT NULL,
	punto_kilometrico numeric(8, 3) NOT NULL,
	urgencia varchar(20) NOT NULL,
	estado varchar(20) DEFAULT 'ABIERTA'::character varying NOT NULL,
	fecha_creacion timestamp DEFAULT CURRENT_TIMESTAMP NOT NULL,
	fecha_actualizacion timestamp DEFAULT CURRENT_TIMESTAMP NOT NULL,
	fecha_cierre timestamp NULL,
	latitud numeric(9, 6) NULL,
	longitud numeric(9, 6) NULL,
	precision_gps_metros numeric(8, 2) NULL,
	CONSTRAINT incidencias_estado_check CHECK (((estado)::text = ANY ((ARRAY['ABIERTA'::character varying, 'ASIGNADA'::character varying, 'EN_REVISION'::character varying, 'RESUELTA'::character varying, 'CERRADA'::character varying])::text[]))),
	CONSTRAINT incidencias_pkey PRIMARY KEY (id_incidencia),
	CONSTRAINT incidencias_urgencia_check CHECK (((urgencia)::text = ANY ((ARRAY['BAJA'::character varying, 'MEDIA'::character varying, 'ALTA'::character varying, 'CRITICA'::character varying])::text[]))),
	CONSTRAINT fk_incidencias_lineas FOREIGN KEY (id_linea) REFERENCES public.lineas(id_linea),
	CONSTRAINT fk_incidencias_operario FOREIGN KEY (id_operario_creador) REFERENCES public.usuarios(id_usuario),
	CONSTRAINT fk_incidencias_vias FOREIGN KEY (id_via) REFERENCES public.vias(id_via)
);
CREATE INDEX idx_incidencias_estado ON public.incidencias USING btree (estado);
CREATE INDEX idx_incidencias_fecha_creacion ON public.incidencias USING btree (fecha_creacion);
CREATE INDEX idx_incidencias_linea_via ON public.incidencias USING btree (id_linea, id_via);
CREATE INDEX idx_incidencias_urgencia ON public.incidencias USING btree (urgencia);
